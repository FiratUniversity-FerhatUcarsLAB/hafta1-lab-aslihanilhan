        MEVCUT_RENK.G = MEVCUT_RENK.G + ADIM_BÜYÜKLÜĞÜ  // Yeşili Artır
    e. DEĞİLSE EĞER MEVCUT_RENK.G > HEDEF_RENK.G İSE:
        MEVCUT_RENK.G = MEVCUT_RENK.G - ADIM_BÜYÜKLÜĞÜ  // Yeşili Azalt
    f. MEVCUT_RENK.G'yi (0, MAKS_DEĞER) arasında TUT.

    // --- Mavi Bileşeni Ayarla (B) ---
    g. EĞER MEVCUT_RENK.B < HEDEF_RENK.B İSE:
        MEVCUT_RENK.B = MEVCUT_RENK.B + ADIM_BÜYÜKLÜĞÜ  // Maviyi Artır
